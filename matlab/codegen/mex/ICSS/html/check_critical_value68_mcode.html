140function [exceeds, position] = check_critical_value(Dk, D_star)
141% CHECK_CRITICAL_VALUE Check if the max value of a range exceeds the
142% critcal
143%   Check if M over the range of Dk exceeds D_star (default: 1.358), 
144%   if so, it is at -position-
145
146    if nargin < 2
147        D_star = 1.358;
148    end
149
150    [value, position] = max(abs(Dk));
151    M = (sqrt(length(Dk)/2) * value);
152    exceeds = M > D_star;
153end
