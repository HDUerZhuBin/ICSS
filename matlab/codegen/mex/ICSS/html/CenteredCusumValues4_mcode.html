1function [ Dk, Ck ] = CenteredCusumValues( data )
 2%CENTERED_CUSUM_VALUES Calculate the centered and normalized cumulative sum of squared
 3%   Based on 'Use of Cumulative Sums of Squares for Retrospective Detection
 4%   of Changes of Variance', by Inclan and Tiao, 1994
 5%
 6%   input: the data the process
 7%   ouput: 
 8%       - the centerd and normalised cumulative sums,
 9%       - the cumulative sums,
10%       - the position for which |Dk| is max
11
12    squared = data.^2;
13    Ck = cumsum(squared);
14    CT = Ck(end);
15    T = length(data);
16    ks = 1:T;
17    
18    Dk = Ck./CT - (ks/T);
19end
